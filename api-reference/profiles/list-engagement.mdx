---
title: 'List engagement profiles'
openapi: 'GET /profiles/engagement'
---

List all engagement-only profiles in your workspace. Engagement profiles can only be used for auto-engagement features like auto-likes, auto-comments, and auto-reposts.

## Response

Returns an array of engagement profile objects.

<ResponseField name="id" type="string">
  UUID of the profile
</ResponseField>

<ResponseField name="name" type="string">
  Display name of the profile
</ResponseField>

<ResponseField name="detail" type="string">
  Additional profile details (e.g., username, handle)
</ResponseField>

<ResponseField name="channel" type="string">
  Social media channel: `LinkedIn` or `Twitter`
</ResponseField>

<ResponseField name="profileImageUrl" type="string">
  URL to the profile's avatar image
</ResponseField>

<ResponseField name="isReintegrationRequired" type="boolean">
  Whether the profile needs to be reconnected
</ResponseField>

<ResponseField name="createdAt" type="datetime">
  Timestamp when the profile was connected
</ResponseField>

<ResponseField name="createdBy" type="object">
  User who connected the profile
  
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      User UUID
    </ResponseField>
    
    <ResponseField name="email" type="string">
      User email address
    </ResponseField>
    
    <ResponseField name="firstName" type="string">
      User first name
    </ResponseField>
    
    <ResponseField name="lastName" type="string">
      User last name
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl --request GET \
  --url https://app.tryordinal.com/api/v1/profiles/engagement \
  --header 'Authorization: Bearer YOUR_API_KEY'
```

```javascript JavaScript
const response = await fetch(
  'https://app.tryordinal.com/api/v1/profiles/engagement',
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const profiles = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://app.tryordinal.com/api/v1/profiles/engagement',
    headers={'Authorization': 'Bearer YOUR_API_KEY'}
)

profiles = response.json()
```
