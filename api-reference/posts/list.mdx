---
title: 'List posts'
openapi: 'GET /posts'
---

Retrieve a paginated list of posts in your workspace.

## Query parameters

<ParamField query="limit" type="number" default={10}>
  Number of posts to return (1-100)
</ParamField>

<ParamField query="offset" type="number" default={0}>
  Number of posts to skip for pagination
</ParamField>

## Response

<ResponseField name="posts" type="array">
  Array of post objects
  
  <Expandable title="post object">
    <ResponseField name="id" type="string">
      UUID of the post
    </ResponseField>
    
    <ResponseField name="title" type="string">
      The post title
    </ResponseField>
    
    <ResponseField name="channels" type="array">
      Array of channels this post is configured for
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Current status: `Tentative`, `ToDo`, `InProgress`, `ForReview`, `Blocked`, `Finalized`, `Scheduled`, or `Posted`
    </ResponseField>
    
    <ResponseField name="publishDate" type="string">
      Scheduled publish date (YYYY-MM-DD) in workspace timezone
    </ResponseField>
    
    <ResponseField name="publishAt" type="datetime">
      Scheduled publish timestamp (UTC). Null if no specific time is set.
    </ResponseField>
    
    <ResponseField name="createdAt" type="datetime">
      Timestamp when the post was created
    </ResponseField>
    
    <ResponseField name="updatedAt" type="datetime">
      Timestamp when the post was last updated
    </ResponseField>
    
    <ResponseField name="linkedIn" type="object">
      LinkedIn content (if configured)
      
      <Expandable title="properties">
        <ResponseField name="copy" type="string">
          The post text content
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl --request GET \
  --url 'https://app.tryordinal.com/api/v1/posts?limit=20&offset=0' \
  --header 'Authorization: Bearer YOUR_API_KEY'
```

```javascript JavaScript
const response = await fetch(
  'https://app.tryordinal.com/api/v1/posts?limit=20&offset=0',
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://app.tryordinal.com/api/v1/posts',
    headers={'Authorization': 'Bearer YOUR_API_KEY'},
    params={'limit': 20, 'offset': 0}
)

data = response.json()
```

</RequestExample>
