---
title: 'Create post'
openapi: 'POST /posts'
---

Create a new post in your workspace. Posts can be created for multiple channels and scheduled for publishing.

## Request body

<ParamField body="title" type="string" required>
  The title of the post
</ParamField>

<ParamField body="status" type="enum" required>
  The status of the post. One of: `Tentative`, `ToDo`, `InProgress`, `ForReview`, `Blocked`, `Finalized`, `Scheduled`, `Posted`
</ParamField>

<ParamField body="labelIds" type="array">
  Array of label UUIDs to attach to the post
</ParamField>

<ParamField body="campaignId" type="string">
  UUID of the campaign to associate with this post
</ParamField>

<ParamField body="publishAt" type="datetime">
  Scheduled publish timestamp in ISO 8601 format. If not provided, the post will not be scheduled.
</ParamField>

<ParamField body="notes" type="string">
  Internal notes about the post
</ParamField>

### LinkedIn content

<ParamField body="linkedIn" type="object">
  LinkedIn-specific content configuration
  
  <Expandable title="properties">
    <ParamField body="profileId" type="string" required>
      UUID of the LinkedIn profile to post from
    </ParamField>
    
    <ParamField body="copy" type="string" required>
      The post text content
    </ParamField>
    
    <ParamField body="mentions" type="array">
      Array of LinkedIn mentions
      
      <Expandable title="mention object">
        <ParamField body="username" type="string" required>
          LinkedIn username to mention
        </ParamField>
        
        <ParamField body="type" type="enum" required>
          Type of mention: `User` or `Organization`
        </ParamField>
        
        <ParamField body="showFirstNameOnly" type="boolean" default={false}>
          Show only first name in mention
        </ParamField>
      </Expandable>
    </ParamField>
    
    <ParamField body="linkPreview" type="object">
      Link preview configuration
      
      <Expandable title="properties">
        <ParamField body="url" type="string" required>
          URL to preview
        </ParamField>
        
        <ParamField body="title" type="string" required>
          Preview title (1-400 characters)
        </ParamField>
        
        <ParamField body="description" type="string">
          Preview description (max 4086 characters)
        </ParamField>
        
        <ParamField body="thumbnailId" type="string">
          UUID of the thumbnail asset
        </ParamField>
      </Expandable>
    </ParamField>
    
    <ParamField body="poll" type="object">
      Poll configuration
      
      <Expandable title="properties">
        <ParamField body="question" type="string" required>
          Poll question (1-140 characters)
        </ParamField>
        
        <ParamField body="options" type="array" required>
          Poll options (2-4 options)
          
          <Expandable title="option object">
            <ParamField body="id" type="string" required>
              UUID for the option
            </ParamField>
            
            <ParamField body="text" type="string" required>
              Option text (1-30 characters)
            </ParamField>
          </Expandable>
        </ParamField>
        
        <ParamField body="duration" type="enum" required>
          Poll duration: `ONE_DAY`, `THREE_DAYS`, `SEVEN_DAYS`, or `FOURTEEN_DAYS`
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="id" type="string">
  UUID of the created post
</ResponseField>

<ResponseField name="url" type="string">
  URL to view the post in Ordinal
</ResponseField>

<ResponseField name="title" type="string">
  The post title
</ResponseField>

<ResponseField name="channels" type="array">
  Array of channels this post is configured for
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the post
</ResponseField>

<ResponseField name="createdAt" type="datetime">
  Timestamp when the post was created
</ResponseField>

<ResponseField name="updatedAt" type="datetime">
  Timestamp when the post was last updated
</ResponseField>
