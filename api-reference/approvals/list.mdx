---
title: 'List post approvals'
openapi: 'GET /posts/{postId}/approvals'
---

List all approval requests for a specific post.

## Path parameters

<ParamField path="postId" type="string" required>
  UUID of the post
</ParamField>

## Response

<ResponseField name="approvals" type="array">
  Array of approval objects
  
  <Expandable title="approval object">
    <ResponseField name="id" type="string">
      UUID of the approval
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Approval status: `Requested` or `Approved`
    </ResponseField>
    
    <ResponseField name="isBlocking" type="boolean">
      Whether this approval blocks post scheduling
    </ResponseField>
    
    <ResponseField name="message" type="string">
      Optional message attached to the approval request
    </ResponseField>
    
    <ResponseField name="dueDate" type="datetime">
      Optional due date for the approval
    </ResponseField>
    
    <ResponseField name="approvedAt" type="datetime">
      Timestamp when approved (null if not yet approved)
    </ResponseField>
    
    <ResponseField name="createdAt" type="datetime">
      Timestamp when the approval was requested
    </ResponseField>
    
    <ResponseField name="user" type="object">
      User who needs to approve
      
      <Expandable title="properties">
        <ResponseField name="id" type="string">
          User UUID
        </ResponseField>
        
        <ResponseField name="firstName" type="string">
          User first name
        </ResponseField>
        
        <ResponseField name="lastName" type="string">
          User last name
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="requestedBy" type="object">
      User who requested the approval
      
      <Expandable title="properties">
        <ResponseField name="id" type="string">
          User UUID
        </ResponseField>
        
        <ResponseField name="firstName" type="string">
          User first name
        </ResponseField>
        
        <ResponseField name="lastName" type="string">
          User last name
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
