---
title: 'Create approval requests'
openapi: 'POST /approvals'
---

Create one or more approval requests for a post.

## Request body

<ParamField body="postId" type="string" required>
  UUID of the post to request approvals for
</ParamField>

<ParamField body="approvals" type="array" required>
  Array of approval request objects
  
  <Expandable title="approval object">
    <ParamField body="userId" type="string" required>
      UUID of the user to request approval from
    </ParamField>
    
    <ParamField body="message" type="string">
      Optional message to include with the approval request
    </ParamField>
    
    <ParamField body="dueDate" type="datetime">
      Optional due date for the approval
    </ParamField>
    
    <ParamField body="isBlocking" type="boolean" default={false}>
      Whether this approval blocks post scheduling
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="createdApprovals" type="array">
  Array of newly created approval objects
</ResponseField>

<ResponseField name="existingApprovals" type="array">
  Array of approvals that already existed for these users
</ResponseField>
